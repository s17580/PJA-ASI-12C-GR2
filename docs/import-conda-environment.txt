# Anaconda:	2024.02-1
# Conda:	conda 24.3.0

---------------------------------------------------------------------------------
---------------------------------------------------------------------------------
02. Add conda-forge as the highest priority channel
---------------------------------------------------------------------------------

conda config --add channels conda-forge

---------------------------------------------------------------------------------
---------------------------------------------------------------------------------
03. Activate strict channel priority
---------------------------------------------------------------------------------

conda config --describe channel_priority
conda config --set channel_priority strict

---------------------------------------------------------------------------------
---------------------------------------------------------------------------------
04. Import conda environment
---------------------------------------------------------------------------------

# Check
conda env list

# Import
conda env create -f C:\a\PJA-ASI-12C-GR2.yaml -n PJA-ASI-12C-GR2

---------------------------------------------------------------------------------
---------------------------------------------------------------------------------
05. Activate environment
---------------------------------------------------------------------------------

conda activate PJA-ASI-12C-GR2
conda info --envs

---------------------------------------------------------------------------------
---------------------------------------------------------------------------------
06. Check packages
---------------------------------------------------------------------------------

conda list -n PJA-ASI-12C-GR2
